Toute deviation peut avoir des consequences juridiques ou financieres.4.1 Calcul des points// 1 pt par tranche de 10 EUR — SANS PLAFONDfunction calculerPoints(montantCourse) { return Math.floor(montantCourse / 10); // Exemples : 107 EUR = 10 pts | 200 EUR = 20 pts | 36 EUR = 3 pts}4.2 Tarification vehiculesfunction calculerMontant(distanceKm, typeVehicule) { const tarifs = { berline: { forfait: 12.00, seuil: 6, prix_km: 2.00 }, van: { forfait: 12.00, seuil: 6, prix_km: 3.00 } }; const t = tarifs[typeVehicule]; if (distanceKm &lt;= t.seuil) return t.forfait; const total = t.forfait + (distanceKm - t.seuil) * t.prix_km; return parseFloat(total.toFixed(2)); // Toujours 2 decimales}4.3 Niveaux Ambassadeurfunction calculerNiveau(points) { if (points &gt;= 5000) return &quot;black&quot;; if (points &gt;= 2000) return &quot;elite&quot;; if (points &gt;= 500) return &quot;pro&quot;; return &quot;starter&quot;;}4.4 Expiration bons — Horodatage a la minute exactefunction calculerExpiration(remisAt, validiteMois) { const expiration = new Date(remisAt); expiration.setMonth(expiration.getMonth() + validiteMois); return expiration; // Remis le 11/05/2026 a 14h35 + 3 mois = expire le 11/08/2026 a 14h35}4.5 Triple verification QR code fournisseurasync function validerBonQR(tokenQR, codeSecret, fournisseurId) { const bon = await getBonByToken(tokenQR); if (!bon) return { succes: false, raison: &quot;token_invalide&quot; }; // Verif 1 — Code secret correct ? const codeOk = await bcrypt.compare(codeSecret, fournisseur.code_secret_hash); if (!codeOk) { await incrementerTentatives(fournisseurId); if (tentatives &gt;= 3) await bloquerValidation(tokenQR); return { succes: false, raison: &quot;code_incorrect&quot; }; } // Verif 2 — QR appartient a ce fournisseur ? if (bon.fournisseur_id !== fournisseurId) return { succes: false, raison: &quot;mauvais_fournisseur&quot; }; // Verif 3 — Prestation = offre active de ce fournisseur ? const offre = await getOffre(bon.offre_id); if (offre.fournisseur_id !== fournisseurId || offre.statut !== &quot;en_ligne&quot;) return { succes: false, raison: &quot;offre_invalide&quot; }; // Verif 4 — Non expire ? if (new Date() &gt; new Date(bon.expire_at)) return { succes: false, raison: &quot;expire&quot; }; // Verif 5 — Non utilise ? if (bon.statut === &quot;utilise&quot;) return { succes: false, raison: &quot;deja_utilise&quot; }; // Tout OK → Valider await marquerBonUtilise(bon.id); await declencherPaiementFournisseur(bon); return { succes: true };}4.6 Sanctions differees silencieusesasync function appliquerSanction(ambassadeurId, points, motif, courseId) { const amb = await getAmbassadeur(ambassadeurId); if (amb.points_solde &gt;= points) { // Solde suffisant : prelevement + notification simultanement await preleverpPoints(ambassadeurId, points); await notifier(ambassadeurId, `-${points} pts preleves (${motif})`); } else { // Solde insuffisant : sanction differee SILENCIEUSEMENT // NE PAS notifier l&apos;Ambassadeur await db.sanctions_en_attente.insert({ ambassadeur_id: ambassadeurId, points, motif, course_id: courseId, statut: &apos;en_attente&apos; }); }}// Appeler a chaque gain de pointsasync function verifierSanctionsEnAttente(ambassadeurId) { const sanctions = await db.sanctions_en_attente.findAll({ where: { ambassadeur_id: ambassadeurId, statut: &apos;en_attente&apos; } }); for (const s of sanctions) { const amb = await getAmbassadeur(ambassadeurId); if (amb.points_solde &gt;= s.points) { await preleverpPoints(ambassadeurId, s.points); await db.sanctions_en_attente.update({ statut: &apos;execute&apos;, execute_at: new Date() }); await notifier(ambassadeurId, `-${s.points} pts preleves (${s.motif})`); } }}5.
